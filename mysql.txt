To clear screen from mysql prompt in ubuntu terminal
------------------------------------------------------
\! clear
======================================================
To connect local mysql database from ubuntu terminal
------------------------------------------------------
mysql -u root -p 
=======================================================
To show databases
-------------------------------------------------------
SHOW DATABASES;
=======================================================
To know current user
-------------------------------------------------------
SELECT current_user();
=======================================================
To know all users in mysql
-------------------------------------------------------
SELECT USER FROM mysql.user;
=======================================================
To drop user 
-------------------------------------------------------
DROP USER username;
=======================================================
To Create user
-------------------------------------------------------
CREATE USER siri IDENTIFIED BY 'password'
=======================================================
To Grant permissions to the user
-------------------------------------------------------
GRANT ALL PRIVILEGES ON database_name.* TO 'username'@'localhost';
=======================================================
To Create Database
-------------------------------------------------------
CREATE DATABASE db;
